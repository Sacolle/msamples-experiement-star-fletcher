** Projeto Experimental

Queremos agora observar se há uma distribuição bimodal, ou até trimodal nos blocos
Então usamos um tamanho maior de bloco base -> 360, porém com menos iterações e menos combinações.


#+BEGIN_SRC R :session *R* :async yes
  library(DoE.base)
  library(glue)
  library(tidyverse)

  border_size = 4
  absorb_size = 8
  total_width = 360
  width = total_width - 2 * (border_size + absorb_size)
  seed = 0
  output = 5 # output every 5 steps
  replications = 3

  block_sizes = c(1000, 3375, 8000, 27000, 64000) |> map(function(x){ total_width / x^(1/3) })

  generate_experiment <- function(factors, replications, seed) {
    DoE.base::fac.design(
      nfactors = length(factors),
      replications = replications,
      repeat.only = FALSE,
      randomize = TRUE,
      seed = seed,
      nlevels = sapply(factors, length),
      factor.names = factors
    )
  }

  cpu_factors <- list(
    WithIO = c(0, 1),
    Segmentation = block_sizes
  )

  # machines = c("poti", "cei", "draco")


  ex <- generate_experiment(cpu_factors, replications, seed)

  print(dim(ex))

  ex |> readr::write_csv(glue("./modal-distribution.csv"))
#+END_SRC
